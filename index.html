GET '/articles' '/articles/index' display a list of all articles
Path Template: 
get '/articles' do

							#your code here

						end
 Suggestions: (to be taken with a grain of salt) 
get '/articles' do
						
						@articles = Article.all #define instance variable for view
						
						erb :'/articles/index' #shows all articles view (index)
						
						end

POST '/articles' '/articles/create' create new articles
Path Template: 
post '/articles' do

							#your code here

						end
 Suggestions: (to be taken with a grain of salt) 
post '/articles' do

							#below works with properly formatted params in html form
							@article = Article.new(params[:article]) #create new article

							if @article.save #saves new article or returns false if unsuccessful
							redirect '/articles' #links back to articles index page
							else 
							erb :errors #shows an errors view you define
							end

						end

						end
 Suggestions: (to be taken with a grain of salt) 
put '/articles/:id' do

						#get params from url
						@article = Article.find(params[:id]) #define variable to edit

						@article.assign_attributes(params[:article]) #assign new attributes

						if @article.save #saves new article or returns false if unsuccessful
						redirect '/articles' #links back to articles index page
						else 
						erb :errors #shows an errors view you define
						end

						end
